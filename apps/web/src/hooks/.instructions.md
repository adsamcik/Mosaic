## Hook Design Patterns

### Naming Convention
- Start with `use` prefix
- File: `kebab-case.ts` (e.g., `use-photos.ts`)
- Export: `camelCase` (e.g., `usePhotos`)

### Structure Template

```typescript
import { useState, useCallback, useEffect } from 'react';

interface UseFeatureOptions {
  initialValue?: string;
  onSuccess?: () => void;
}

interface UseFeatureReturn {
  data: string | null;
  isLoading: boolean;
  error: Error | null;
  doAction: (input: string) => Promise<void>;
}

export function useFeature(options: UseFeatureOptions = {}): UseFeatureReturn {
  const { initialValue = null, onSuccess } = options;
  
  // State
  const [data, setData] = useState<string | null>(initialValue);
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  
  // Actions (memoized)
  const doAction = useCallback(async (input: string) => {
    setIsLoading(true);
    setError(null);
    try {
      const result = await apiCall(input);
      setData(result);
      onSuccess?.();
    } catch (err) {
      setError(err instanceof Error ? err : new Error('Unknown error'));
    } finally {
      setIsLoading(false);
    }
  }, [onSuccess]);
  
  return { data, isLoading, error, doAction };
}
```

### Return Type Pattern

Always return an object (not tuple) for extensibility:

```typescript
// ✅ Good: Object return (extensible)
return { photos, isLoading, error, refetch };

// ❌ Bad: Tuple return (not extensible)
return [photos, isLoading, error];
```

### Explicit Return Types

Always define return interface:

```typescript
// ✅ Good: Explicit return type
interface UsePhotosReturn {
  photos: Photo[];
  isLoading: boolean;
  error: Error | null;
}

export function usePhotos(albumId: string): UsePhotosReturn {
  // ...
}

// ❌ Bad: Inferred return type
export function usePhotos(albumId: string) {
  // TypeScript infers, but consumers can't easily see the shape
}
```

## Data Fetching Hooks

Use React Query patterns for server state:

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';

export function useAlbums() {
  return useQuery({
    queryKey: ['albums'],
    queryFn: fetchAlbums,
    staleTime: 5 * 60 * 1000, // 5 minutes
  });
}

export function useCreateAlbum() {
  const queryClient = useQueryClient();
  
  return useMutation({
    mutationFn: createAlbum,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['albums'] });
    },
  });
}
```

## Feature Logic Hooks

Encapsulate complex UI logic:

```typescript
export function useSelection<T extends { id: string }>() {
  const [selectedIds, setSelectedIds] = useState<Set<string>>(new Set());
  
  const toggle = useCallback((id: string) => {
    setSelectedIds(prev => {
      const next = new Set(prev);
      if (next.has(id)) {
        next.delete(id);
      } else {
        next.add(id);
      }
      return next;
    });
  }, []);
  
  const selectAll = useCallback((items: T[]) => {
    setSelectedIds(new Set(items.map(item => item.id)));
  }, []);
  
  const clear = useCallback(() => {
    setSelectedIds(new Set());
  }, []);
  
  const isSelected = useCallback((id: string) => selectedIds.has(id), [selectedIds]);
  
  return {
    selectedIds,
    selectedCount: selectedIds.size,
    toggle,
    selectAll,
    clear,
    isSelected,
  };
}
```

## Crypto Hooks

Crypto hooks wrap the crypto worker for React components:

```typescript
import { useMemo } from 'react';
import { getCryptoWorker } from '../workers/crypto-worker-client';

export function useEpochKeys(albumId: string) {
  const cryptoWorker = useMemo(() => getCryptoWorker(), []);
  
  const decryptPhoto = useCallback(async (
    encryptedData: Uint8Array,
    epochId: number
  ): Promise<Uint8Array> => {
    const epochKey = await cryptoWorker.getEpochKey(albumId, epochId);
    try {
      return await cryptoWorker.decryptShard(encryptedData, epochKey);
    } finally {
      // Key cleanup happens in worker
    }
  }, [albumId, cryptoWorker]);
  
  return { decryptPhoto };
}
```

## Testing Hooks

Use `@testing-library/react` with `renderHook`:

```typescript
import { renderHook, act } from '@testing-library/react';
import { useSelection } from './useSelection';

describe('useSelection', () => {
  it('should toggle selection', () => {
    const { result } = renderHook(() => useSelection());
    
    act(() => {
      result.current.toggle('photo-1');
    });
    
    expect(result.current.isSelected('photo-1')).toBe(true);
    expect(result.current.selectedCount).toBe(1);
  });
  
  it('should clear all selections', () => {
    const { result } = renderHook(() => useSelection());
    
    act(() => {
      result.current.toggle('photo-1');
      result.current.toggle('photo-2');
      result.current.clear();
    });
    
    expect(result.current.selectedCount).toBe(0);
  });
});
```

## Anti-Patterns

```typescript
// ❌ Bad: Side effects in render
export function usePhotos() {
  const [photos, setPhotos] = useState([]);
  fetchPhotos().then(setPhotos); // Runs every render!
  return photos;
}

// ✅ Good: Use useEffect for side effects
export function usePhotos() {
  const [photos, setPhotos] = useState([]);
  useEffect(() => {
    fetchPhotos().then(setPhotos);
  }, []);
  return photos;
}

// ❌ Bad: Unstable callbacks
export function useFeature() {
  const doSomething = () => { /* ... */ }; // New function every render
  return { doSomething };
}

// ✅ Good: Memoized callbacks
export function useFeature() {
  const doSomething = useCallback(() => { /* ... */ }, []);
  return { doSomething };
}

// ❌ Bad: Exposing internal state setters
export function useFeature() {
  const [data, setData] = useState(null);
  return { data, setData }; // Consumers can break invariants
}

// ✅ Good: Expose actions, not setters
export function useFeature() {
  const [data, setData] = useState(null);
  const updateData = useCallback((newData) => {
    // Validate and transform
    setData(sanitize(newData));
  }, []);
  return { data, updateData };
}
```
