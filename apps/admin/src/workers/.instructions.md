## Worker Architecture

```
Main Thread (React)
    │
    ├── Comlink Proxy ──► Crypto Worker (libsodium)
    │                     └── Encryption/decryption, key derivation
    │
    ├── Comlink Proxy ──► DB Worker (SQLite-WASM)
    │                     └── Local database operations
    │
    └── Comlink Proxy ──► Geo Worker
                          └── GPS coordinate processing
```

## Strict Constraints

### Main Thread Isolation

```typescript
// ✅ Good: Crypto operations in worker
const cryptoWorker = wrap<CryptoWorker>(new Worker(...));
const encrypted = await cryptoWorker.encrypt(data, key);

// ❌ FORBIDDEN: Crypto in main thread
import sodium from 'libsodium-wrappers-sumo';
const encrypted = sodium.crypto_secretbox_easy(data, nonce, key);
```

### Memory Safety (Crypto Worker)

```typescript
// ✅ REQUIRED: Zero sensitive buffers
export async function deriveKey(password: string): Promise<Uint8Array> {
  const masterKey = sodium.crypto_pwhash(...);
  try {
    return hkdf(masterKey, context);
  } finally {
    sodium.memzero(masterKey);  // CRITICAL
  }
}

// ❌ FORBIDDEN: Keys left in memory
export async function deriveKey(password: string): Promise<Uint8Array> {
  const masterKey = sodium.crypto_pwhash(...);
  return hkdf(masterKey, context);  // masterKey leaked!
}
```

### No Sensitive Logging

```typescript
// ✅ Allowed: Log metadata
console.log('Encrypting shard', { shardIndex, size: data.length });

// ❌ FORBIDDEN: Log key material
console.log('Using key:', key);
console.log('Decrypted:', sodium.to_hex(plaintext));
```

## Comlink Patterns

### Worker Definition

```typescript
// crypto.worker.ts
import { expose } from 'comlink';
import sodium from 'libsodium-wrappers-sumo';

const cryptoWorker = {
  async init(): Promise<void> {
    await sodium.ready;
  },
  
  async encrypt(data: Uint8Array, key: Uint8Array): Promise<Uint8Array> {
    if (key.length !== 32) throw new Error('Invalid key length');
    const nonce = sodium.randombytes_buf(24);
    const ciphertext = sodium.crypto_aead_xchacha20poly1305_ietf_encrypt(
      data, null, null, nonce, key
    );
    // Prepend nonce to ciphertext
    const result = new Uint8Array(24 + ciphertext.length);
    result.set(nonce);
    result.set(ciphertext, 24);
    return result;
  },
  
  async decrypt(envelope: Uint8Array, key: Uint8Array): Promise<Uint8Array> {
    const nonce = envelope.slice(0, 24);
    const ciphertext = envelope.slice(24);
    return sodium.crypto_aead_xchacha20poly1305_ietf_decrypt(
      null, ciphertext, null, nonce, key
    );
  },
};

export type CryptoWorker = typeof cryptoWorker;
expose(cryptoWorker);
```

### Client Wrapper

```typescript
// crypto-worker-client.ts
import { wrap, type Remote } from 'comlink';
import type { CryptoWorker } from './crypto.worker';

let cryptoWorker: Remote<CryptoWorker> | null = null;

export function getCryptoWorker(): Remote<CryptoWorker> {
  if (!cryptoWorker) {
    const worker = new Worker(
      new URL('./crypto.worker.ts', import.meta.url),
      { type: 'module' }
    );
    cryptoWorker = wrap<CryptoWorker>(worker);
  }
  return cryptoWorker;
}

// Usage in React components/hooks
const worker = getCryptoWorker();
await worker.init();
const encrypted = await worker.encrypt(data, key);
```

### Transferable Objects

For large binary data, use `Comlink.transfer()` to avoid copying:

```typescript
import { transfer } from 'comlink';

// ✅ Good: Transfer large buffers
const result = await worker.processImage(
  transfer(imageBuffer, [imageBuffer.buffer])
);

// ❌ Bad: Copy large buffers (slow)
const result = await worker.processImage(imageBuffer);
```

## DB Worker Patterns

### SQLite-WASM Setup

```typescript
// db.worker.ts
import { expose } from 'comlink';
import initSqlJs, { type Database } from 'sql.js';

let db: Database | null = null;

const dbWorker = {
  async init(): Promise<void> {
    const SQL = await initSqlJs({
      locateFile: (file) => `/sql-wasm/${file}`,
    });
    // Use OPFS for persistence
    db = new SQL.Database();
  },
  
  async exec(sql: string, params?: unknown[]): Promise<void> {
    if (!db) throw new Error('Database not initialized');
    db.run(sql, params);
  },
  
  async query<T>(sql: string, params?: unknown[]): Promise<T[]> {
    if (!db) throw new Error('Database not initialized');
    const stmt = db.prepare(sql);
    stmt.bind(params);
    const results: T[] = [];
    while (stmt.step()) {
      results.push(stmt.getAsObject() as T);
    }
    stmt.free();
    return results;
  },
};

export type DbWorker = typeof dbWorker;
expose(dbWorker);
```

## Error Handling

Workers should throw typed errors that can cross the Comlink boundary:

```typescript
// In worker
export class DecryptionError extends Error {
  constructor(message: string, public readonly code: string) {
    super(message);
    this.name = 'DecryptionError';
  }
}

// Throw from worker method
if (authFailed) {
  throw new DecryptionError('Authentication failed', 'AUTH_FAILED');
}

// In main thread
try {
  await cryptoWorker.decrypt(data, key);
} catch (error) {
  if (error instanceof Error && error.name === 'DecryptionError') {
    // Handle decryption-specific error
  }
}
```

```typescript
// crypto.worker.test.ts
import { cryptoWorker } from './crypto.worker';

describe('CryptoWorker', () => {
  beforeAll(async () => {
    await cryptoWorker.init();
  });
  
  it('should encrypt and decrypt data', async () => {
    const key = new Uint8Array(32).fill(1);
    const plaintext = new Uint8Array([1, 2, 3, 4, 5]);
    
    const encrypted = await cryptoWorker.encrypt(plaintext, key);
    const decrypted = await cryptoWorker.decrypt(encrypted, key);
    
    expect(decrypted).toEqual(plaintext);
  });
});
```

## Performance Considerations

### Worker Startup
- Workers are initialized lazily on first use
- Keep singleton pattern for worker instances
- Initialize critical workers early (on app mount)

### Message Size
- Use `transfer()` for buffers > 1MB
- Batch small operations to reduce message overhead
- Consider streaming for very large files

### Thread Pool
- Browser limits concurrent workers (~4-8 depending on cores)
- Reuse worker instances, don't create per-operation
- Queue operations during high load
